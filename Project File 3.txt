n = int(input("Enter number of Calculation"))

count = 1

while (count <= n):
  num1 = float(input("Enter first number"))
  num2 = float(input("Enter second number"))

  print("Choose operation")
  print("1, Addition")
  print("2, Subtraction")
  print("3, Mulpiplication")
  print("4, Division")

  choice = int(input("Enter your choice (1,2,3,4)"))

  if choice == 1:
     result = num1 + num2
     print("Result:", result)

  elif choice == 2:
     result = num1 - num2
     print("Result:", result)

  elif choice == 3:
     result = num1 * num2
     print("Result:", result)

  elif choice == 4:
     if num2 == 0:

        result = num1 / num2
        print("Result:", result)
     else:
        print("Enter Division by zero")


  else:
    print("Invalid choice")

  count = count + 1
